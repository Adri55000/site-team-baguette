# --- HTTP (port 80) ---

    listen 80;
    server_name teambaguette.duckdns.org;

    # ‚úÖ ACME challenge pour Certbot (renouvellement)
    location ^~ /.well-known/acme-challenge/ {
        root /var/www/certbot;
        default_type "text/plain";
        try_files $uri =404;
    }

    # üîÅ Redirection vers HTTPS
    return 301 https://$host$request_uri;

# --- HTTPS (port 443) ---
server {
    listen 443 ssl;
    server_name teambaguette.duckdns.org;

    client_max_body_size 10M;

    # ‚úÖ ACME challenge aussi ici (robuste)
    location ^~ /.well-known/acme-challenge/ {
        root /var/www/certbot;
        default_type "text/plain";
        try_files $uri =404;
    }

    # ‚úÖ Proxy vers Gunicorn
    location / {
        proxy_pass http://127.0.0.1:8000;

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # ‚úÖ Recommand√© pour SSE (indices)
        proxy_http_version 1.1;
        proxy_buffering off;
        proxy_read_timeout 3600;
    }

    # --- Certbot SSL (laisse tel quel) ---
    ssl_certificate     /etc/letsencrypt/live/teambaguette.duckdns.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/teambaguette.duckdns.org/privkey.pem;
    include             /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam         /etc/letsencrypt/ssl-dhparams.pem;
}
